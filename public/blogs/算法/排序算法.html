# 选择排序

最简单但是最没用(时间复杂度高，不稳定)的排序算法，也有优化的空间

func sort(){
	arr:=[]int{5,3,6,8,1,7,9,4,2}
	for i:=0;i<len(arr)-1;i++{
		var minPos=i
		for j:=i+1;j<len(arr);j++{
			if arr[j]<arr[minPos]{
				minPos=j
			}
		}
		temp:=arr[i]
		arr[i]=arr[minPos]
		arr[minPos]=temp

	}
	for i:=0;i<len(arr);i++{
		fmt.Println(arr[i])
	}
}

# 冒泡排序

func sort(a []int){
	for i:=len(a)-1;i>0;i--{
		for j:=0;j<i;j++{
			if a[j]>a[j+1]{
				a[j],a[j+1]=a[j+1],a[j]
			}
		}
	}
}

# 插入排序

对基本有序的数组最好用

func sort(a []int){
	for i:=1;i<len(a);i++{
		for j:=i;j>0;j--{
			if a[j]<a[j-1]{
				a[j],a[j-1]=a[j-1],a[j]
			}
		}
	}
}
